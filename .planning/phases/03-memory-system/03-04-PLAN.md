   ```typescript
     private static instance: MemoryOrchestratorImpl | null = null;
     private isExtracting: boolean = false;
     private lastExtractionTime: number = 0;
     private readonly EXTRACTION_COOLDOWN = 60000; // 1 minute between extractions

     static getInstance(): MemoryOrchestratorImpl { ... }
   }
   ```

3. Implement extractAndStore method:
   ```typescript
   async extractAndStore(): Promise<void> {
     // Guard against concurrent extractions
     if (this.isExtracting) {
       console.log('[MemoryOrchestrator] Extraction already in progress');
       return;
     }

     // Cooldown check
     const now = Date.now();
     if (now - this.lastExtractionTime < this.EXTRACTION_COOLDOWN) {
       console.log('[MemoryOrchestrator] Cooldown active, skipping');
       return;
     }

     // Check LLM availability
     if (!LLMService.isReady()) {
       console.log('[MemoryOrchestrator] LLM not ready, skipping');
       return;
     }

     // Get conversation
     const { messages } = useChatStore.getState();
     if (messages.length < 2) {
       console.log('[MemoryOrchestrator] Too few messages, skipping');
       return;
     }

     this.isExtracting = true;
     this.lastExtractionTime = now;

     try {
       console.log('[MemoryOrchestrator] Starting extraction...');

       const conversationText = formatConversationForExtraction(messages);
       const extracted = await extractMemories(conversationText);

       if (extracted.length > 0) {
         useMemoryStore.getState().addMemories(extracted);
         console.log('[MemoryOrchestrator] Stored', extracted.length, 'memories');
       } else {
         console.log('[MemoryOrchestrator] No memories extracted');
       }
     } catch (error) {
       console.error('[MemoryOrchestrator] Extraction failed:', error);
     } finally {
       this.isExtracting = false;
     }
   }
   ```

4. Implement prune method (optional cleanup):

   ```typescript
   prune(): void {
     const THRESHOLD = 0.1; // Remove memories with <10% relevance
     useMemoryStore.getState().pruneDecayed(THRESHOLD);
   }
   ```

5. Export singleton:

   ```typescript
   import { useEffect, useRef } from 'react';
   ```

2. Create useConversationEnd hook:
   ```typescript
   export function useConversationEnd(): void {
     const appState = useRef(AppState.currentState);
     const { markConversationEnded, messages } = useChatStore();

     useEffect(() => {
       const subscription = AppState.addEventListener(
         'change',
         (nextAppState: AppStateStatus) => {
           // App going to background
           if (
             appState.current === 'active' &&
             nextAppState.match(/inactive|background/)
           ) {
             console.log('[useConversationEnd] App backgrounding, triggering extraction');

             // Mark conversation ended (for metadata)
             markConversationEnded();

             // Trigger extraction (non-blocking)
             MemoryOrchestrator.extractAndStore().catch((err) => {
               console.error('[useConversationEnd] Extraction error:', err);
             });
           }

           appState.current = nextAppState;
         }
       );

       return () => {
         subscription.remove();
       };
     }, [markConversationEnded]);
   }
   ```

The hook:
- Listens to AppState changes
- When app goes from active to background/inactive:
  * Marks conversation as ended (for chatStore metadata)
  * Triggers extraction (fire-and-forget, catches errors)
- Does NOT block the app lifecycle
- Uses ref to track previous state (necessary for detecting transition)

Note: This hook should be used in the root ChatScreen or App component to ensure it's always active during chat.
